Write a C program to find the Nth term in the fibonacci series. 

For example:

Input	Result
0        0
1        1
4        3




#include<stdio.h>
int main(){
    int n,f0=0,f1=1,fn=0;
    scanf("%d",&n);
    if(n==0 || n==1){
        printf("%d",n);
    }else{
    for(int i=0;i<n-1;i++){
        fn=f0+f1;
        f0=f1;
        f1=fn;
        }
    
    printf("%d",fn);
    }
}
